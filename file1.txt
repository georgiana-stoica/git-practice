 const [bonds, setBonds] = useState([]);

  useEffect( () =>{
    getBondsFromAPI();},
    []
  );

  const getBondsFromAPI = () => {
    getAllBonds()
  .then(res => {
      setBonds(res.data);
  })
  .catch(err => {
      setBonds([]);
      console.log(err);
  })
  }

  return (
    <Row>
    {bonds.map(bond=>(
        <div className='container' key={bond.isin}>
        <BondsDetail info={bond}  />

        </div>
    ))}
    </Row>
  
    )
}

export default AllBonds
