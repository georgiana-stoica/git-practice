import 'bootstrap/dist/css/bootstrap.min.css';
import React, { useState } from "react"; // Fix the import statement for useState
import './App.css';
import SignIn from './components/SignIn';
import FinancialDashboard from './components/FinancialDashboard';
import { Route, Routes, Navigate } from 'react-router-dom'; // Import Navigate to handle redirection

function App() {
  const [isLoggedIn, setIsLoggedIn] = useState(false); // State to track login status

  // Function to handle successful login and set isLoggedIn to true
  const handleLogin = () => {
    setIsLoggedIn(true);
  };

  return (
    <>
      <Routes>
        {/* If isLoggedIn is false, render the SignIn component */}
        <Route path="/login" element={<SignIn onLogin={handleLogin} />} />

        {/* If isLoggedIn is true, render the FinancialDashboard component, otherwise, redirect to the login page */}
        {isLoggedIn ? (
          <Route path="/dashboard" element={<FinancialDashboard />} />
        ) : (
          <Navigate to="/login" />
        )}
      </Routes>
    </>
  );
}
