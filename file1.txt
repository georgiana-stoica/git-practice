import React, { useState, useEffect } from 'react';
import axios from 'axios';

const Maturity = () => {
  const [selectedDate, setSelectedDate] = useState('');
  const [bonds, setBonds] = useState([]);

  const handleDateChange = (event) => {
    setSelectedDate(event.target.value);
  };

  useEffect(() => {
    // Fetch bond data from the API based on the selected date
    const fetchBonds = async () => {
      try {
        const response = await axios.get(`http://localhost:8080/bonds/maturity/workday/date?dateStr=${selectedDate}`);
        setBonds(response.data.map(bond => ({ ...bond, maturityDate: formatDate(bond.maturityDate) })));
      } catch (error) {
        console.error(error);
      }
    };

    if (selectedDate) {
      fetchBonds();
    }
  }, [selectedDate]);

  // Helper function to format timestamp to 'YYYY-MM-DD' date string
  const formatDate = (timestamp) => {
    const date = new Date(timestamp);
    const year = date.getFullYear();
    const month = String(date.getMonth() + 1).padStart(2, '0');
    const day = String(date.getDate()).padStart(2, '0');
    return `${year}-${month}-${day}`;
  };

  return (
    <div>
      <label htmlFor="date">Select Date:</label>
      <input type="date" id="date" value={selectedDate} onChange={handleDateChange} />
      {bonds.length > 0 ? (
        bonds.map((bond) => <div key={bond.ISIN}>{bond.ISIN}</div>)
      ) : (
        <div>No bonds due for maturity on the selected date.</div>
      )}
    </div>
  );
};

export default Maturity;
